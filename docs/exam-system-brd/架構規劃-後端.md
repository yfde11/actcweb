**實作邏輯**：
```javascript
// services/examGeneration.js（新增）
async function generateFromBank(examId, domainRatio) {
    const exam = await Exam.findById(examId);
    const questionRefs = [];

    for (const [domain, percentage] of Object.entries(domainRatio)) {
        const count = Math.round(exam.questionsPerAttempt * (percentage / 100));
        
        // 從題庫隨機抽取該 domain 的題目（排除已引用的）
        const questions = await Question.aggregate([
            { $match: { 
                domain: parseInt(domain), 
                examIds: { $ne: examId }  // 排除已引用
            }},
            { $sample: { size: count } }
        ]);

        questions.forEach(q => questionRefs.push(q._id));
    }

    // 更新 Exam 的 questionRefs
    exam.questionRefs = questionRefs;
    await exam.save();
}
```

### 4.2 輸入驗證

```javascript
// 所有 API 輸入必須驗證
const { body, validationErrors } = req;

// domain 驗證（1-8）
if (body.domain && ![1,2,3,4,5,6,7,8].includes(body.domain)) {
    return errorResponse(res, 400, 'VALIDATION_ERROR', 'Domain must be 1-8');
}

// domainRatio 驗證（總和 = 100）
if (body.domainRatio) {
    const total = Object.values(body.domainRatio).reduce((a, b) => a + b, 0);
    if (Math.abs(total - 100) > 1) {
        return errorResponse(res, 400, 'VALIDATION_ERROR', 'Domain ratio must sum to 100%');
    }
}
```

## 5. 效能優化

### 5.1 索引策略

| 欄位 | 索引類型 | 原因 |
|-------|----------|------|
| `domain` + `difficulty` | 複合索引 | 按 domain + 難度篩選 |
| `examIds` | 陣列索引 | 查找被哪些考試引用 |
| `domain` + `createdAt` | 複合索引 | 按 domain 排序 |

### 5.2 查詢優化

```javascript
// 使用 aggregation pipeline 提升效能
const questions = await Question.aggregate([
    { $match: { domain: 1, difficulty: 'easy' } },
    { $sample: { size: 20 } },  // MongoDB 3.6+ 隨機抽取
    { $project: {  // 只回傳必要欄位
        content: 1,
        type: 1,
        options: 1,
        correctOptionIndex: 1
    }}
]);
```

### 5.3 分頁與緩存

```javascript
// 分頁
const page = parseInt(req.query.page) || 1;
const limit = parseInt(req.query.limit) || 20;
const skip = (page - 1) * limit;

const [questions, total] = await Promise.all([
    Question.find(query).skip(skip).limit(limit),
    Question.countDocuments(query)
]);
```

## 7. 架構決策記錄 (ADR)

### ADR-001：採用題庫模式（多對多）

**狀態**：Accepted

**上下文**：
現有「考試專屬題目」設計無法複用題目，導致重複建立相同題目。

**決策**：
將 Question 模型改為「題庫模式」，新增 `examIds`（多對多）和 `domain`（CISSP 8 domains）。

**後果**：
- ✅ 題目可複用（被多個考試引用）
- ✅ 支援題目複用率統計
- ❌ 需修改現有 200+ 題目資料結構
- ❌ 開發時間增加 30%

**取捨**：
- 維持現狀（考試專屬題目）→ 無法複用題目
- 購買第三方題庫服務 → 成本高、無法客製
